Your config file should look like this:

[Graphics]
Fullscreen=0
ColorDepth=32
ResolutionX=800
ResolutionY=600
MaximumFramesPerSecond=60

[Network]
Server1=poorserver.no-ip.org

[Media]
MediaFile=evidyon.gme

[Audio]
Enable=0

[Debugging]
f0=1
f1=1
f2=1
f3=1
f4=1
f5=1
f6=1
f7=1
f8=1
f9=1
